Takes a list of names and returns a list of the 4 letter names.

{% highlight python %}

#example input: ['Timmy','john','stacy','pete','elizabeth']

def is_friend(x):
   names = []
   for name in x:
      if len(name) == 4:
       names.append(name)
   return names

 #example output: ['john','pete']

{% endhighlight %}
